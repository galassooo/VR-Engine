82 bool init()
83 {
84 vr::EVRInitError error = vr::VRInitError_None;
85
86 // Init VR system:
87 std::cout << "Using OpenVR " << vr::k_nSteamVRVersionMajor << "." << vr::k_nSteamVRVersionMinor << "." << vr::k_nSteamVRVersionBuild << std::endl;
88 vrSys = vr::VR_Init(&error, vr::VRApplication_Scene);
89 if (error != vr::VRInitError_None)
90 {
91 vrSys = nullptr;
92 std::cout << "[ERROR] Unable to init VR runtime: " << vr::VR_GetVRInitErrorAsEnglishDescription(error) << std::endl;
93 return false;
94 }
95
96 // Init render models:
97 vrModels = (vr::IVRRenderModels*)vr::VR_GetGenericInterface(vr::IVRRenderModels_Version, &error);
98 if (vrModels == nullptr)
99 {
100 vrSys = nullptr;
101 vr::VR_Shutdown();
102 std::cout << "[ERROR] Unable to get render model interface: " << vr::VR_GetVRInitErrorAsEnglishDescription(error) << std::endl;
103 return false;
104 }
105
106 // Initialize the compositor:
107 vrComp = vr::VRCompositor();
108 if (!vrComp)
109 {
110 vrModels = nullptr;
111 vrSys = nullptr;
112 vr::VR_Shutdown();
113 std::cout << "[ERROR] Unable to get VR compositor" << std::endl;
114 return false;
115 }
116
117 // Tweaks:
118 //vrComp->ShowMirrorWindow();
119
120 // Init controller tracking:
121 controllers.clear();
122 for (unsigned int c = 0; c < vr::k_unMaxTrackedDeviceCount; c++)
123 {
124 if (vrSys->GetTrackedDeviceClass(c) == vr::TrackedDeviceClass_Controller)
125 {
126 std::cout << " Found controller at " << c << std::endl;
127 Controller* cont = new Controller();
128 cont->id = c;
129
130 unsigned int bufferLen = vrSys->GetStringTrackedDeviceProperty(c, vr::Prop_RenderModelName_String, nullptr, 0, nullptr);
131 if (bufferLen == 0)
132 {
133 std::cout << "[ERROR] Unable to get controller render model" << std::endl;
134 delete cont;
135 continue;
136 }
137
138 std::string result;
139 result.resize(bufferLen);
140 vrSys->GetStringTrackedDeviceProperty(c, vr::Prop_RenderModelName_String, &result[0], bufferLen, nullptr);
141 std::cout << " Controller render model: '" << result << "'" << std::endl;
142 controllers.push_back(cont);
143 }
144 }
145
146 // Done:
147 return true;
148 }

275 static glm::mat4 ovr2ogl(const vr::HmdMatrix34_t& matrix)
276 {
277 return glm::mat4(matrix.m[0][0], matrix.m[1][0], matrix.m[2][0], 0.0f,
278 matrix.m[0][1], matrix.m[1][1], matrix.m[2][1], 0.0f,
